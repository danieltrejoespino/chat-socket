

CREATE TABLE TBL_MSG (
  `ID_MSG` int(11) NOT NULL AUTO_INCREMENT,
  `ID_USER` int(11) NOT NULL,
  `MSG` varchar(1000) NOT NULL,
  `DATE_MSG` date DEFAULT CURRENT_DATE,
  PRIMARY KEY (`ID_MSG`)
) 


CREATE TABLE TBL_USER (
  ID_USER int(11) NOT NULL AUTO_INCREMENT,
  ID_PERFIL int(11) NOT NULL,
  NAME_USER varchar(200) NOT NULL,
  PASS_USER varchar(200) NOT NULL,
  STATUS_USER int(11) NOT NULL,
  PRIMARY KEY (ID_USER)
)


CREATE TABLE TBL_MENU (
  ID_MENU int(11) NOT NULL AUTO_INCREMENT,  
  NAME_MENU varchar(200) NOT NULL,
  STATUS_MENU int(11) NOT NULL,
  PRIMARY KEY (ID_MENU)
)



CREATE TABLE TBL_PHONE_EXT (
  ID_EXT int(10) NOT NULL AUTO_INCREMENT,  
  OWNER_EXT varchar(50) NOT NULL,  
  AREA_EXT varchar(50) NOT NULL,
  NAME_EXT varchar(200) NOT NULL,  
  STATUS_EXT int(1) NOT NULL,
  PRIMARY KEY (ID_EXT)
)
